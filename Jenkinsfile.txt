pipeline {
    agent any
	
    stages {
        stage('Build') {
            steps {
                checkout scm
                echo 'Building..'
                sh 'make'
                archiveArtifacts artifacts: '**/target/*.jar', fingerprint: true
            }
        }
        stage('Test') {
            steps {
                echo 'Testing..'
                sh 'make check || true' 
                junit '**/target/*.xml' 
            }
        }
        stage('Deploy') {
            when {
                expression {
                    currentBuild.result == null || currentBuild.result == 'SUCCESS' ¢Ù
                }
            }
            steps {
                echo 'Deploying..'
                sh 'make publish'
            }
        }
    }
}
